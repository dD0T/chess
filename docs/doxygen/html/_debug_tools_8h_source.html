60 GameState generateRandomState(size_t maxTurns, Rng& rng) {
61  std::uniform_int_distribution<size_t> dst(0, maxTurns);
62  const size_t turnCount = dst(rng);
63 
64  GameState gs;
65  for (size_t i = 0; i < turnCount; ++i) {
66  auto turns = gs.getTurnList();
67  auto turn = random_selection(turns, rng);
68  if (turn == end(turns)) break;
69  gs.applyTurn(*turn);
70  }
71 
72  return gs;
73 }
