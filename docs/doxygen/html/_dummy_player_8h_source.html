74  virtual std::future<Turn> doMakeTurn(GameState state) override {
75  return postPromise([=]() mutable {
76  // Always choose a random turn after a random amount of time
77  // Obviously can't be aborted
78  auto turns = state.getTurnList();
79  auto turnIt = random_selection(turns, m_rng);
80  Turn turn = (turnIt == std::end(turns)) ? Turn() : (*turnIt);
81  int duration = m_msDist(m_rng);
82 
83  LOG(Logging::debug) << "(" << m_seed << ") "
84  << "Will select " << turn << " after " << duration << " seconds sleep";
85  std::this_thread::sleep_for(std::chrono::seconds(duration));
86  return turn;
87  });
88  }
89 private:
90  const unsigned int m_seed;
