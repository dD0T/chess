211  NegamaxResult search_recurse(TGameState state, size_t depth, const size_t maxDepth, Score alpha, Score beta) {
212  if (m_abort) return{ 0, boost::none };
213 
214  const size_t pliesLeft = maxDepth - depth;
215 
216  if (state.isGameOver() || pliesLeft == 0) {
217  return { state.getScore(depth), boost::none };
218  }
219 
220  const Score initialAlpha = alpha;
221 
222  if (TRANSPOSITION_TABLES_ENABLED) {
223  auto tableEntry = m_transpositionTable.lookup(state.getHash());
224  if (tableEntry && tableEntry->depth >= pliesLeft) {
225  ++m_counters.transpositionTableHits;
226 
227  // Deep enough to use directly
228  if (tableEntry->isExactBound()) {
229  // This is an actual result
230  return { tableEntry->score, tableEntry->turn };
231  } else if (tableEntry->isLowerBound()) {
232  // We have a lower bound, adjust alpha accordingly
233  alpha = std::max(alpha, tableEntry->score);
234  } else {
235  assert(tableEntry->isUpperBound());
236  // Upper bound, adjust beta
237  beta = std::min(beta, tableEntry->score);
238  }
239 
240  if (AB_CUTOFF_ENABLED && alpha >= beta) {
241  // Upper or lower bound known for this position will
242  // trigger an alpha beta cutoff. No need to continue
243  // search.
244  ++m_counters.cutoffs;
245  return { tableEntry->score, tableEntry->turn };
246  }
247  }
248  }
249 
250  NegamaxResult bestResult { MIN_SCORE, boost::none };
251 
252  std::vector<Turn> possibleTurns = state.getTurnList();
253  assert(possibleTurns.size() > 0);
254 
255 
256  std::vector<TGameState> consideredStates(possibleTurns.size(), state);
257  std::vector<Option> consideredOptions;
258 
259  consideredOptions.reserve(possibleTurns.size());
260 
261  for (unsigned int i = 0; i < possibleTurns.size(); ++i) {
262  // Need to apply all states at this point to be able to
263  // use their data for move ordering.
264  consideredStates[i].applyTurn(possibleTurns[i]);
265  consideredOptions.emplace_back(consideredStates[i],
266  possibleTurns[i],
267  estimateScoreFor(consideredStates[i],
268  depth + 1));
269  }
270 
271  if (MOVE_ORDERING_ENABLED) {
272  std::sort(begin(consideredOptions), end(consideredOptions));
273  }
274 
275  for (Option& option: consideredOptions) {
276  Turn& turn = *option.turn;
277  TGameState& newState = *option.state;
278 
279  ++m_counters.nodes;
280 
281  NegamaxResult result = -search_recurse(
282  newState, depth + 1, maxDepth,
283  -beta, -alpha);
284 
285  // Check if we improved upon previous turns
286  if (result > bestResult) {
287  ++m_counters.updates;
288 
289  bestResult = result;
290  bestResult.turn = turn;
291  }
292 
293  alpha = std::max(alpha, result.score);
294 
295  if (AB_CUTOFF_ENABLED && alpha >= beta) {
296  ++m_counters.cutoffs;
297  // Enemy player won't let us reach a better score than
298  // his guaranteed beta score. No use in continuing to
299  // search this position as the results would be discarded
300  // anyways.
301  break;
302  }
303 
304  if (m_abort) return{ 0, boost::none };
305  }
306 
307  if (TRANSPOSITION_TABLES_ENABLED) {
308  assert(bestResult.turn);
309 
310  TranspositionTableEntry entry;
311  entry.score = bestResult.score;
312  entry.turn = *bestResult.turn;
313  entry.hash = state.getHash();
314  entry.depth = pliesLeft; // Our results comes from pliesLeft deep
315 
316  if (bestResult.score <= initialAlpha) {
317  // Opponent might have omitted results with a lower score from
318  // this position meaning this is a upper bound.
319  entry.boundType = TranspositionTableEntry::UPPER;
320  } else if (bestResult.score >= beta) {
321  // We might have omitted results with a higher score from this position
322  // meaning this is a lower bound.
323  entry.boundType = TranspositionTableEntry::LOWER;
324  } else {
325  // No cutoff occured. The result is exact.
326  entry.boundType = TranspositionTableEntry::EXACT;
327  }
328 
329  m_transpositionTable.maybeUpdate(entry);
330  }
331 
332  return bestResult;
333  }

344  Score estimateScoreFor(const TGameState& state, size_t depth) const {
345  // Note: Need to negate results as they are from the enemys POV
346  if (TRANSPOSITION_TABLES_ENABLED) {
347  auto entry = m_transpositionTable.lookup(state.getHash());
348  if (!entry) return -state.getScore(depth);
349  else if (entry->isUpperBound()) return -MIN_SCORE;
350  else return -entry->score;
351  } else {
352  // Greedy estimate based on next move
353  return -state.getScore(depth);
354  }
355  }
